def set_up_project_config(project_config):
  """
  Set configurations to have right model names to load pretrained model and to match our models
  """
  if project_config['base_transformer'].lower() == 'bert':
    transformer_model = project_config['base_transformer']+'-' + project_config['model_size'] + '-uncased'
    model_name = transformer_model + '-epoch-'
  elif project_config['base_transformer'].lower() == 'xlnet':
    transformer_model = project_config['base_transformer']+'-' + project_config['model_size'] + '-cased'
    if project_config['use_biseps']:
      model_name = 'xl-biseps'
      if project_config['use_multi_obj']:
        model_name += '-mo'
      elif project_config['use_multi_task']:
        model_name += '-mt'
    else:
      model_name = transformer_model
    model_name += '-epoch-'
  project_config['transformer_model'], project_config['model_name'] = transformer_model, model_name
  return project_config

def augment_data(data):
  """For the multi objective model, we add a new sample for each first sentence.
    Ex: In the original data we have the samples
    - A cat is on a roof    | A cat is outside          | entailement
    - A cat is on a roof    | A cat is inside           | contradition
    - A lady wears a dress  | A lady is wearing clothes | neutral
    - A lady wears a dress  | A man wears a hat         | neutral

    Then we will add 2 new samples
    - A cat is on a roof    | A lady is wearing clothes | neutral
    - A lady wears a dress  | A cat is outside          | neutral

    Every sample in the original dataset will have is_linked = 1 and 
    the new samples will have a target is_linked = 0
  """
  num_train_sample = data.shape[0]
  #Keep track of original data
  data['sample_id'] = range(num_train_sample)
  
  sentence_2_id_per_sentence_1 = data.groupby(by='sentence1')['sample_id'].apply(list).to_dict()

  num_sentence_1 = list(range(len(sentence_2_id_per_sentence_1.keys())))
  np.random.shuffle(num_sentence_1)
  sentences_2_id = list(sentence_2_id_per_sentence_1.values())
  new_sentences_2_id = [sentences_2_id[i][0] for i in num_sentence_1] #First sentence2 for every sentence1

  sentences2 = data.sentence2
  new_data = {'sentence1':list(sentence_2_id_per_sentence_1.keys()),'sentence2':[]}
  new_data['sentence2'] = [sentences2[sentence_2_id] for sentence_2_id in new_sentences_2_id]

  df_new_data = pd.DataFrame(new_data)
  df_new_data['link'] = 0
  df_new_data['id'] = -1
  df_new_data['label1']='neutral' # Randomly picked there is very little chance the sentences are not neutral
  df_new_data.head()

  data['link'] = 1
  data = pd.concat([data,df_new_data])

  data = data.sort_values(by='sentence1').drop('sample_id',axis=1)
  data = data.reset_index(drop=True)
  return data

def preprocess_data(data, model_name, project_config, is_train_data=True, verbose=False, task='SNLI'):
  if 'xlnet' in model_name:
    sep_token, cls_token, pad_token = '<sep>','<cls>','<pad>'
    tokenizer = XLNetTokenizer
  elif 'bert' in model_name:
    tokenizer = BertTokenizer
    sep_token, cls_token, pad_token = '[SEP]','[CLS]','[PAD]'
  else:
    raise ValueError("Model must have BERT or XLNet as base transformer")

  if verbose:
    print('Preprocessing with train data: {} \n With the model: {} \n On task: {} \n Tokens are: {} {} {} \n'.format(is_train_data,model_name,task,sep_token, cls_token, pad_token))

  # Add a column with the tokens added
  data['sentence'] = data['sentence1'].str.lower() +' '+ sep_token +' '+ data['sentence2'].str.lower() + ' ' +sep_token
  if 'xlnet' in model_name:
    data['sentence'] = data['sentence'] +' '+cls_token
  else:
    data['sentence'] = cls_token + ' ' + data['sentence']

  # Prepare XLNet tokenizer
  tokenizer = tokenizer.from_pretrained(model_name, do_lower_case=False, sep_token=sep_token)
  # Tokenize the sentences
  if task == 'SNLI':
    # Some sentences have a '.' at the end, some doesn't, in order to harmonize the sentences, we remove the points
    sentences_tokenized = list(data['sentence'].apply(lambda sentence: list(filter(lambda token: token != '.', tokenizer.tokenize(sentence)))))
  elif task == 'STS':
    sentences_tokenized = list(data['sentence'].apply(lambda sentence: list(tokenizer.tokenize(sentence))))

  if verbose:
    # What is the max size of inputs?
    print("The max length of sentences is:", max([len(sentence) for sentence in sentences_tokenized]))
  MAX_LEN = 128

  # Remove sentences with size above 128 in the STS data
  if task == 'STS':
    len_STS = len(sentences_tokenized)
    indices_removed = [i for i, s in enumerate(sentences_tokenized) if len(s) > MAX_LEN]
    sentences_tokenized = [s for s in sentences_tokenized if len(s) <= MAX_LEN]
    if verbose:
      print(len_STS - len(sentences_tokenized), 'values were removed in the data')

  # Compute attention masks: 1 where there is information, 0 where there is <pad>
  attention_masks = [[1]*len(s) + [0]*(MAX_LEN-len(s)) for s in sentences_tokenized]

  # Correspondance for segments
  segments = { 'sentenceA' : 0, 'sentenceB' : 1, 'sep': 2, 'cls': 3, 'pad': 4 }
  # Compute segments
  SEP_index = [ sentence.index(sep_token) for sentence in sentences_tokenized ]

  segment_ids = [ [segments['sentenceA']]*(SEP_index[i]) \
                + [segments['sep']] \
                + [segments['sentenceB']]*(len(sentences_tokenized[i])-SEP_index[i]-3) \
                + [segments['sep']] \
                + [segments['cls']] \
                + [segments['pad']]*(MAX_LEN-len(sentences_tokenized[i])) \
                for i in range(len(sentences_tokenized)) ]
  
  # Add <pad> until MAX_LEN is reached
  sentences_tokenized = [ sentence + [pad_token]*(MAX_LEN - len(sentence)) for sentence in sentences_tokenized ]
  # Convert tokens to ids in the XLNet vocabulary
  sentences_tokenized_ids = [ tokenizer.convert_tokens_to_ids(tokenized_sentence) for tokenized_sentence in sentences_tokenized ]

  if verbose:
    print('Example \n---> {} \nAfter tokenisation and padding \n---> {} \nToken _ids after \n---> {}'.format(data['sentence'].iloc[0],sentences_tokenized[0],sentences_tokenized_ids[0]))



  # Transform text labels to index
  if is_train_data:
    if task == 'SNLI':
      data_labels = list(data['label1'])
      labels_num = [project_config['labels'].index(label) for label in data_labels]
    elif task == 'STS':
      labels_num = [label for i, label in enumerate(list(data['score']/5.0)) if i not in indices_removed]
      print('STS dataset size: {}'.format(len(labels_num)))

  # transform to tensors
  if is_train_data:
    return torch.tensor(sentences_tokenized_ids), torch.tensor(attention_masks), torch.tensor(segment_ids), torch.tensor(labels_num)
  return torch.tensor(sentences_tokenized_ids), torch.tensor(attention_masks), torch.tensor(segment_ids)

class SequenceCustomSummary(nn.Module):
    """ Compute a single vector summary of a sequence hidden states according to various possibilities:
        Args of the config class:
            summary_type:
                - 'last' => [default] take the last token hidden state (like XLNet)
                - 'seq_concat' => take the two sep tokens hidden states
            summary_use_proj: Add a projection after the vector extraction
            cls_proj: Add a projection after the cls token --> For the multi objective version
            cls_last_dropout: Add dropout after the cls projection

            summary_proj_to_labels: If True, the projection outputs to config.num_labels classes (otherwise to hidden_size). Default: False.
            summary_activation: 'tanh' => add a tanh activation to the output, Other => no activation. Default
            summary_first_dropout: Add a dropout before the projection and activation
            summary_last_dropout: Add a dropout after the projection and activation
            
    """
    def __init__(self, config):
        super(SequenceCustomSummary, self).__init__()

        self.summary_type = config.summary_type if hasattr(config, 'summary_use_proj') else 'last'

        self.summary = nn.Identity()
        if hasattr(config, 'summary_use_proj') and config.summary_use_proj:
            if hasattr(config, 'summary_proj_to_labels') and config.summary_proj_to_labels and config.num_labels > 0:
                num_classes = config.num_labels
            else:
                num_classes = config.hidden_size
            self.summary = nn.Linear(config.hidden_size, num_classes)
        
        if hasattr(config, 'summary_use_concat_proj') and config.summary_use_concat_proj:
            num_classes = config.hidden_size
            self.summary = nn.Linear(2*config.hidden_size, num_classes)

        self.activation = nn.Identity()
        if hasattr(config, 'summary_activation') and config.summary_activation == 'tanh':
            self.activation = nn.Tanh()

        self.first_dropout = nn.Identity()
        if hasattr(config, 'summary_first_dropout') and config.summary_first_dropout > 0:
            self.first_dropout = nn.Dropout(config.summary_first_dropout)

        self.last_dropout = nn.Identity()
        if hasattr(config, 'summary_last_dropout') and config.summary_last_dropout > 0:
            self.last_dropout = nn.Dropout(config.summary_last_dropout)

        self.multi_obj = False
        # Multi objective layers
        if hasattr(config, 'cls_proj'):
          self.multi_obj = True
          self.cls_proj = nn.Linear(config.hidden_size,config.hidden_size)
        if hasattr(config, 'cls_last_dropout') :
          if config.cls_last_dropout > 0:
            self.cls_dropout = nn.Dropout(config.cls_last_dropout)
          else:
            self.cls_dropout = nn.Identity()
        

    def forward(self, hidden_states, seq_indices=None):
        if self.summary_type == 'last':
            output = hidden_states[:, -1]
        elif self.summary_type == 'seq_concat':
            output = torch.reshape(hidden_states[seq_indices],(hidden_states.shape[0],2,hidden_states.shape[2])).view(-1,2*hidden_states.shape[2])

        output = self.first_dropout(output)
        output = self.summary(output)
        output = self.activation(output)
        output = self.last_dropout(output)

        if self.multi_obj:
          cls_output = hidden_states[:, -1]
          cls_output = self.cls_dropout(output)
          cls_output = self.cls_proj(output)
          cls_output = self.activation(output)
          return output, cls_output
        else:
          return output

class XLNetBiSEPs(XLNetPreTrainedModel):
  def __init__(self, config):
    super(XLNetBiSEPs, self).__init__(config)
    self.num_labels = config.num_labels

    self.transformer = XLNetModel(config)
    self.sequence_summary = SequenceCustomSummary(config) # https://github.com/huggingface/transformers/blob/32e1332acf6fd1ad372b81c296d43be441d3b0b1/pytorch_transformers/modeling_utils.py#L643-L644
    self.logits_proj = nn.Linear(config.d_model, config.num_labels)

    if config.use_multi_task:
      self.sts_proj = nn.Linear(config.d_model, 1)

    if config.use_multi_obj:
      self.cls_proj = nn.Linear(config.d_model, 1)
      self.lambda_cls = config.lambda_cls

    self.init_weights()

  def forward(self, input_ids, attention_mask=None, mems=None, perm_mask=None, target_mapping=None,
              token_type_ids=None, input_mask=None, head_mask=None, labels=None, task=None,  cls_label=None):
    transformer_outputs = self.transformer(input_ids,
                                            attention_mask=attention_mask,
                                            mems=mems,
                                            perm_mask=perm_mask,
                                            target_mapping=target_mapping,
                                            token_type_ids=token_type_ids,
                                            input_mask=input_mask,
                                            head_mask=head_mask)
    output = transformer_outputs[0]
    seq_indices = (input_ids==4) #The sep indices

    output = self.sequence_summary(output, seq_indices)
    if task=='inference':
      if cls_label is not None:
        #Multi objective configuration
        output, cls_output = output[0],output[1]
        cls_logit = self.cls_proj(cls_output)
      logits = self.logits_proj(output)
    else:
      logits = self.sts_proj(output)

    outputs = (logits,) + transformer_outputs[1:]  # Keep mems, hidden states, attentions if there are in it

    if labels is not None:
      if task == 'inference':
        loss_fct = nn.CrossEntropyLoss()
        loss = loss_fct(logits.view(-1, self.num_labels), labels.view(-1)) 
        if cls_label is not None:
          #Multi objective configuration
          pos_weight = torch.ones([1],device=device)
          pos_weight[0]=0.3
          loss_fct_cls = nn.BCEWithLogitsLoss(pos_weight=pos_weight)
          loss_cls = self.lambda_cls*loss_fct_cls(cls_logit.view(-1),cls_label.view(-1))
          outputs = (loss_cls,) + outputs
      else:
        loss_fct = nn.MSELoss()
        loss = loss_fct(logits.view(-1), labels.view(-1))

      outputs = (loss,) + outputs

    return outputs  # return (loss(es) , logits, (mems), (hidden states), (attentions)

def predict_batch(model, batch, verbose, task, is_train_data=True, multi_obj=False, step=None):
  # Unpack the inputs from our dataloader
  if is_train_data:
    if multi_obj:
      b_input_ids, b_input_mask, b_segments, b_links, b_labels = batch
    else:
      b_input_ids, b_input_mask, b_segments, b_labels = batch
      b_links = None
  else:
    if multi_obj:
      b_input_ids, b_input_mask, b_segments, b_links = batch
    else:
      b_input_ids, b_input_mask, b_segments = batch
      b_links = None
    b_labels = None

  # Forward pass
  outputs = model( b_input_ids, token_type_ids=b_segments, attention_mask=b_input_mask, labels=b_labels, task=task, cls_label=b_links )
  if is_train_data:
    loss = outputs[0]
    if multi_obj:
      cls_loss = outputs[1]
      logits = outputs[2]
    else: 
      logits = outputs[1]
      cls_loss=None
    if verbose:
      print('{} : {} loss {}'.format(step, loss.item(), task))
      if multi_obj:
        print('{} : {} loss {}'.format(step, cls_loss.item(), 'is_linked loss'))
    return loss, cls_loss, logits, b_labels
  else:
    logits = outputs[0]
    return logits

def fit(model, optimizer, train_dataloader, project_config, sts_dataloader=None, save=False, verbose=True, break_step=-1, n_epoch=1):
  train_loss_set = [[] for i in range(n_epoch)]

  for e in trange(n_epoch, desc="Epoch"):
    # Training
    tr_loss, nb_tr_steps = 0,0

    model.train()

    # Create iterator dataloaders
    train_dataloader_iterator = iter(train_dataloader)
    num_batch = len(train_dataloader_iterator)
    if project_config["use_multi_task"] and sts_dataloader is not None:
      train_dataloader_iterator_STS = iter(sts_dataloader)
      num_batch += len(train_dataloader_iterator_STS)
      sts_remaining = len(train_dataloader_iterator_STS)
    else:
      sts_remaining = 0

    for step in range(num_batch):      
      if project_config["use_multi_task"] and step%10 == 0 and sts_remaining > 0:
        # Use a similarity batch to train every 10 batches, until there is no similarity data remaining
        task = 'similarity'
        sts_remaining -= 1
        batch = next(train_dataloader_iterator_STS)
      else:
        task = 'inference'
        batch = next(train_dataloader_iterator)

      # Send data to gpu
      batch = tuple(t.to(device) for t in batch)
      
      # Clear out the gradients (by default they accumulate)
      optimizer.zero_grad()

      loss, cls_loss, logits, _ = model.predict_batch(model, batch, verbose, task, step=step, multi_obj=project_config['use_multi_obj'])

      train_loss_set[e].append(loss.item())
      # Backward propagation
      if project_config['use_multi_obj']:
        loss = loss+cls_loss
      loss.backward()
      optimizer.step()

      if task == 'inference':
        tr_loss += loss.item()
        nb_tr_steps += 1
      if step==break_step:
        break
    
    if nb_tr_steps>0:
      print("\nTrain loss: {}".format(tr_loss/nb_tr_steps))
    
    if save:
      print('Saving model at {}'.format(project_config['model_name']+str(project_config['load_epoch']+e+1)+'.pt'))
      torch.save(model.state_dict(), gdrive_path + project_config['model_name']+str(project_config['load_epoch']+e+1)+'.pt')
  return train_loss_set

def predict(model, data_loader, project_config, is_train_data=True):
  # Validation

  model.eval()
  eval_loss, eval_accuracy = 0, 0
  nb_eval_steps, nb_eval_examples = 0, 0
  logits_full,groud_truth = [], []

  for batch in data_loader:
    batch = tuple(t.to(device) for t in batch)

    with torch.no_grad():

def prepare_data(project_config):
  train_SNLI, test_SNLI, train_STS = pd.read_csv('snli_train.csv'),pd.read_csv('snli_test.csv'),pd.read_csv('sts-b_train.csv')
  if project_config['use_multi_obj']:
    train_SNLI = augment_data(train_SNLI)

  train_SNLI, validation_SNLI = split_train_validation(train_SNLI, project_config['validation_size'])
  # We don't really need validation for the second task, as it's not our primary task
  train_STS = train_STS

  # Preprocess the data
  print('SNLI \n')
  train_SNLI_sentences_tokenized_ids, train_SNLI_attention_masks, train_SNLI_segment_ids, train_SNLI_labels_num = preprocess_data(train_SNLI, project_config['transformer_model'], project_config, is_train_data=True, verbose=False, task='SNLI')
  validation_SNLI_sentences_tokenized_ids, validation_SNLI_attention_masks, validation_SNLI_segment_ids, validation_SNLI_labels_num = preprocess_data(validation_SNLI, project_config['transformer_model'], project_config,is_train_data=True, verbose=False, task='SNLI')
  test_SNLI_sentences_tokenized_ids, test_SNLI_attention_masks, test_SNLI_segment_ids = preprocess_data(test_SNLI, project_config['transformer_model'], project_config, is_train_data=False, verbose=False, task='SNLI')

  print('\nSTS \n')
  train_STS_sentences_tokenized_ids, train_STS_attention_masks, train_STS_segment_ids, train_STS_labels_num = preprocess_data(train_STS, project_config['transformer_model'], project_config, is_train_data=True, verbose=False, task='STS')

  if project_config['use_multi_obj']:
    train_links, validation_links = torch.tensor(train_SNLI['link'].values,dtype=torch.float32), torch.tensor(validation_SNLI['link'].values,dtype=torch.float32)
    test_links = torch.ones([len(test_SNLI)])
  else: 
    train_links, validation_links, test_links = None, None, None

  # Define the batch_size according to the size of the model

  if project_config['model_size']=='large':
    batch_size = 16
  elif project_config['model_size']=='base' :
    batch_size = 32

  # Create the DataLoaders

  train_SNLI_dataloader = get_dataloader(train_SNLI_sentences_tokenized_ids, train_SNLI_attention_masks, train_SNLI_segment_ids, links=train_links, labels=train_SNLI_labels_num, batch_size=batch_size)
  validation_dataloader = get_dataloader(validation_SNLI_sentences_tokenized_ids, validation_SNLI_attention_masks, validation_SNLI_segment_ids,is_random=False, links=validation_links, labels=validation_SNLI_labels_num, batch_size=batch_size)
  test_dataloader = get_dataloader(test_SNLI_sentences_tokenized_ids, test_SNLI_attention_masks, test_SNLI_segment_ids, links=test_links, batch_size=batch_size, is_random=False)

  train_STS_dataloader = get_dataloader(train_STS_sentences_tokenized_ids, train_STS_attention_masks, train_STS_segment_ids, labels=train_STS_labels_num, batch_size=batch_size)

  return (train_SNLI_dataloader,validation_dataloader,test_dataloader),train_STS_dataloader

def full_pipeline(project_config,precomputed=False, train_SNLI_dataloader=None, validation_dataloader=None,test_dataloader=None,train_STS_dataloader=None):
  project_config = set_up_project_config(project_config)
  if not precomputed:
    (train_SNLI_dataloader,validation_dataloader,test_dataloader),train_STS_dataloader = prepare_data(project_config)
  model_local,config = get_model(project_config)

  param_optimizer = list(model_local.named_parameters())
  no_decay = ['bias', 'gamma', 'beta']
  optimizer_grouped_parameters = [
    {'params': [p for n, p in param_optimizer if not any(nd in n for nd in no_decay)],
      'weight_decay_rate': 0.01},
    {'params': [p for n, p in param_optimizer if any(nd in n for nd in no_decay)],
      'weight_decay_rate': 0.0}
  ]
  optimizer = AdamW(optimizer_grouped_parameters,
                    lr=2e-5)
  model_local.fit = fit
  model_local.predict_batch = predict_batch
  model_local.predict = predict
  train_loss_set = model_local.fit(model_local, optimizer, train_SNLI_dataloader, project_config, sts_dataloader=train_STS_dataloader, break_step=10)
  test_logits,_ = model_local.predict(model_local, test_dataloader, project_config, is_train_data=False)
  validation_logits, validation_gr = model_local.predict(model_local, validation_dataloader, project_config, is_train_data=True)
  validation_categs = logits_to_category(validation_logits)

  test_labels = logits_to_labels(test_logits, project_config)
  return test_labels, validation_categs, validation_gr
